<h1>Vibe coding deer in the forest</h1>


<div id="game-container">

    <div id="ui">Berries: <span id="score">0</span></div>

    <canvas id="gameCanvas" width="800" height="400"></canvas>

    

    <div id="overlay">

        <h1 style="color: #ff4d6d;">GAME OVER</h1>

        <p id="final-score" style="font-size: 24px; margin-bottom: 30px;"></p>

        <button onclick="resetGame()">RESTART</button>

<script>

    const canvas = document.getElementById('gameCanvas');

    const ctx = canvas.getContext('2d');

    const scoreElement = document.getElementById('score');

    const overlay = document.getElementById('overlay');

    const finalScoreText = document.getElementById('final-score');


    let score = 0;

    let gameActive = true;

    let frames = 0;


    const player = {

        x: 80,

        y: 200,

        w: 60,

        h: 70,

        speed: 7

    };


    let enemies = []; 

    let collectibles = []; 


    const keys = {};

    window.addEventListener('keydown', e => keys[e.code] = true);

    window.addEventListener('keyup', e => keys[e.code] = false);


    function spawnItems() {

        if (frames % 90 === 0) {

            enemies.push({

                x: canvas.width,

                y: Math.random() * (canvas.height - 50),

                w: 90,

                h: 50,

                speed: 5 + (score / 5)

            });

        }

        if (frames % 60 === 0) {

            collectibles.push({

                x: canvas.width,

                y: Math.random() * (canvas.height - 30),

                w: 25,

                h: 25

            });

        }

    }


    function draw99NightsDeer(x, y) {

        // LEG STYLE (Blocky Roblox style)

        ctx.fillStyle = '#4e342e'; // Dark brown legs

        ctx.fillRect(x + 10, y + 50, 8, 20);

        ctx.fillRect(x + 35, y + 50, 8, 20);


        // MAIN BODY (Rectangle)

        ctx.fillStyle = '#6b4226'; // Saddle brown

        ctx.fillRect(x + 5, y + 20, 45, 35);


        // WHITE SPOTS

        ctx.fillStyle = 'rgba(255, 255, 255, 0.6)';

        ctx.fillRect(x + 15, y + 25, 4, 4);

        ctx.fillRect(x + 30, y + 30, 4, 4);

        ctx.fillRect(x + 22, y + 38, 4, 4);


        // NECK

        ctx.fillStyle = '#6b4226';

        ctx.fillRect(x + 35, y + 5, 12, 18);


        // HEAD

        ctx.fillRect(x + 32, y - 8, 25, 18);

        

        // FACE (Tan muzzle area like the Roblox deer)

        ctx.fillStyle = '#d2b48c'; 

        ctx.fillRect(x + 45, y - 5, 14, 12);


        // EYES (Large, dark, staring)

        ctx.fillStyle = 'black';

        ctx.beginPath();

        ctx.arc(x + 50, y - 1, 3.5, 0, Math.PI * 2);

        ctx.fill();


        // ANTLERS (Small dark branches)

        ctx.fillStyle = '#3a261a';

        ctx.fillRect(x + 38, y - 18, 3, 10);

        ctx.fillRect(x + 48, y - 18, 3, 10);

        ctx.fillRect(x + 35, y - 18, 15, 3);

    }


    function drawBerry(x, y) {

        ctx.fillStyle = '#ff4d6d';

        ctx.beginPath();

        ctx.arc(x + 12, y + 12, 10, 0, Math.PI * 2);

        ctx.fill();

        ctx.fillStyle = '#2d6a4f';

        ctx.fillRect(x + 10, y, 4, 6);

    }


    function drawFlyingShark(x, y, w, h) {

        ctx.fillStyle = '#457b9d';

        ctx.beginPath();

        ctx.ellipse(x + w/2, y + h/2, w/2, h/2, 0, 0, Math.PI * 2);

        ctx.fill();

        // Fin

        ctx.beginPath();

        ctx.moveTo(x + w/2, y);

        ctx.lineTo(x + w/2 - 15, y - 20);

        ctx.lineTo(x + w/2 - 30, y);

        ctx.fill();

        // Tail

        ctx.beginPath();

        ctx.moveTo(x + w, y + h/2);

        ctx.lineTo(x + w + 15, y);

        ctx.lineTo(x + w + 15, y + h);

        ctx.fill();

        // Red glowing eye

        ctx.fillStyle = 'red';

        ctx.beginPath();

        ctx.arc(x + 15, y + 15, 4, 0, Math.PI * 2);

        ctx.fill();

    }


    function update() {

        if (!gameActive) return;


        if (keys['ArrowUp'] && player.y > 20) player.y -= player.speed;

        if (keys['ArrowDown'] && player.y < canvas.height - player.h) player.y += player.speed;

        if (keys['ArrowLeft'] && player.x > 0) player.x -= player.speed;

        if (keys['ArrowRight'] && player.x < canvas.width - player.w) player.x += player.speed;


        enemies.forEach((shark, index) => {

            shark.x -= shark.speed;

            if (player.x < shark.x + shark.w &&

                player.x + player.w > shark.x &&

                player.y < shark.y + shark.h &&

                player.y + player.h > shark.y) {

                gameOver();

            }

            if (shark.x + shark.w < 0) enemies.splice(index, 1);

        });


        collectibles.forEach((berry, index) => {

            berry.x -= 3;

            if (player.x < berry.x + berry.w &&

                player.x + player.w > berry.x &&

                player.y < berry.y + berry.h &&

                player.y + player.h > berry.y) {

                score++;

                scoreElement.innerText = score;

                collectibles.splice(index, 1);

            }

            if (berry.x + berry.w < 0) collectibles.splice(index, 1);

        });


        frames++;

        spawnItems();

    }


    function draw() {

        ctx.clearRect(0, 0, canvas.width, canvas.height);

        

        // Dark forest background details

        ctx.fillStyle = '#081c15';

        for(let i=0; i<10; i++) {

            let treeX = ((i * 150) - (frames % 150));

            ctx.fillRect(treeX, 0, 40, 400);

        }


        draw99NightsDeer(player.x, player.y);

        enemies.forEach(s => drawFlyingShark(s.x, s.y, s.w, s.h));

        collectibles.forEach(b => drawBerry(b.x, b.y));

    }


    function gameLoop() {

        update();

        draw();

        if (gameActive) requestAnimationFrame(gameLoop);

    }


    function gameOver() {

        gameActive = false;

        overlay.style.display = 'flex';

        finalScoreText.innerText = `The deer collected ${score} berries.`;

    }


    function resetGame() {

        score = 0;

        scoreElement.innerText = score;

        enemies = [];

        collectibles = [];

        player.y = 200;

        gameActive = true;

        overlay.style.display = 'none';

        gameLoop();

    }


    gameLoop();

</script>
